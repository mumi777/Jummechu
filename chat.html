음식 추천 받기

from js import document, fetch from pyscript import create_proxy import asyncio async def handle_submit(event): text = document.getElementById('textInput').value if text: response = await fetch("/classify", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({"text": text}) }) result = await response.json() document.getElementById('output').innerText = result['result'] # handle_submit 함수를 create_proxy로 감싸서 프록시 생성 handle_submit_proxy = create_proxy(handle_submit) # 버튼 클릭 이벤트에 프록시 함수 연결 document.getElementById('submitBtn').addEventListener('click', handle_submit_proxy)
