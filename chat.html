from js import document, fetch import asyncio async def handle_submit(event): text = document.getElementById('textInput').value if text: response = await fetch("/classify", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({"text": text}) }) result = await response.json() document.getElementById('output').innerText = result['result'] document.getElementById('submitBtn').addEventListener('click', handle_submit)
